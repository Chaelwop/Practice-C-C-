// Java program to print a number triangle pattern import java.util.*; public class GeeksForGeeks { // Function to demonstrate the pattern public static void printPattern(int n) { int i, j; // Outer loop to handle the number of rows for (i = 1; i <= n; i++) { // Inner loop to print spaces for (j = 1; j <= n - i; j++) { System.out.print(" "); } // Inner loop to print numbers for (j = 1; j <= i; j++) { System.out.print(i + " "); } // Print new line for each row System.out.println(); } } // Driven function public static void main(String args[]) { int n = 6; // You can change this value to print a larger or smaller triangle printPattern(n); } }
